## 神经元
```python
import numpy as np

a1 = -0.9
a2 = -0.5
a3 = -0.7

inputs = np.array([a1,a2,a3])
w1 =  0.8
w2 = -0.4
w3 = 0
weights = np.array([w1,w2,w3])
b1 = 0

sum1 = np.dot(inputs, weights)+b1


def activation_ReLu(inputs):
    return np.maximum(0, inputs)


print(sum1)
print(activation_ReLu(sum1))

```
